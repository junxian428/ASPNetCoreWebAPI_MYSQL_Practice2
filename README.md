CREATE TABLE user (
    Id INT PRIMARY KEY,
    Username VARCHAR(255),
    Mail VARCHAR(255),
    PhoneNumber VARCHAR(20),
    Skillsets VARCHAR(255),
    Hobby VARCHAR(255)
);
